118 inline void SliceVolume::setSliceSphereCoord(const QVector3D & coord)
119 {
120  const auto r = coord.x();
121  const auto theta = coord.y();
122  const auto phi = coord.z();
123  const auto sinPhi = std::sin(qDegreesToRadians(phi));
124  const auto x = r * sinPhi * std::sin(qDegreesToRadians(theta)) + 0.5;
125  const auto y = r* sinPhi * std::cos(qDegreesToRadians(theta))+0.5;
126  const auto z = r * std::cos(qDegreesToRadians(phi))+0.5;
127  m_A = x-0.5;
128  m_B = y -0.5;
129  m_C = z -0.5;
130  m_D = -x * (m_A) - y * ( m_B) - z * (m_C);
131 }
